# xenoist faction mechanical events

country_event = {
	id = lnd_faction_events.10
	hide_window = yes
	is_triggered_only = yes

	trigger = { 
		is_country_type = default
		NOT = { has_country_flag = lnd_xenophile_faction_loyality_cooldown }
		NOT = { has_modifier = lnd_xenophile_faction_loyal }
		AND = {
			has_faction = xenoist
			any_pop_faction = {
				is_pop_faction_type = xenoist
				support > 0.1
				faction_approval > 0.59
			}
		}

	 }
	 immediate = {
		add_modifier = {
			modifier = lnd_xenophile_faction_loyal
			days = -1
		}
		set_timed_country_flag = {
			flag = lnd_xenophile_faction_loyality_cooldown
			days = 720
		}
		if = {
			limit = {
				NOT = { has_country_flag = lnd_faction_events_STFU }
			}
			country_event = { id = lnd_faction_events.11 days = 1 }
		}
	 }
}
